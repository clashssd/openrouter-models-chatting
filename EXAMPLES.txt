EXAMPLE 8: CACHING RESPONSES
============================
from core.utils import ResponseCache
import tempfile

# Create cache
with tempfile.NamedTemporaryFile(suffix='.json', delete=False) as f:
    cache_file = f.name

cache = ResponseCache(cache_file)

# Cache response
messages = [{"role": "user", "content": "Hello"}]
cache.set("model_id", messages, "Cached response")

# Retrieve from cache
cached = cache.get("model_id", messages)
print(f"Cached response: {cached}")

# Disable caching
cache.toggle(False)
cached = cache.get("model_id", messages)  # Returns None
print(f"With caching disabled: {cached}")

# Clear cache
cache.clear()


EXAMPLE 9: TOKEN COUNTING
=========================
from core.utils import TokenCounter

counter = TokenCounter()

# Count tokens in text
text = "This is a sample text for token counting"

print(f"Tokens in text: {tokens}")

# Count tokens in messages
messages = [
    {"role": "user", "content": "Hello, how are you?"},
    {"role": "assistant", "content": "I am doing well, thank you for asking!"}
]
total_tokens = counter.count_messages(messages)
print(f"Total tokens in messages: {total_tokens}")

# Estimate cost
cost = counter.estimate_cost(
    input_tokens=100,
    output_tokens=200,
    model="gpt-4"
)
print(f"Estimated cost: ${cost['total_cost']:.4f}")

EXAMPLE 13: RATE LIMITING
=========================
from core.rate_limiter import RateLimiter
import time

limiter = RateLimiter()

# Configure rate limiter
limiter.max_retries = 3
limiter.base_delay = 2.0
limiter.max_delay = 30.0

# Define a function that might fail
def api_call():
    import random
    if random.random() < 0.7:
        raise Exception("Rate limit exceeded")
    return "Success"

try:
    result = limiter.execute_with_retry(
        func=api_call,
        endpoint="chat/completions",
        max_retries=3
    )
    print(f"Result: {result}")
except Exception as e:
    print(f"Failed after retries: {e}")

ADVANCED EXAMPLES

Example A: Batch Processing
---------------------------
from core.openrouter_checker import ApiKeyChecker
import json

api_key = "your_api_key_here"
checker = ApiKeyChecker(api_key)

# Process multiple prompts
prompts = [
    "What is AI?",
    "Explain ML",
    "Define NLP"
]

results = {}
for prompt in prompts:
    try:
        response = checker.chat_completion(
            model_id="nvidia/nemotron-3-ultra-550b-a55b:free",
            prompt=prompt
        )
        results[prompt] = response
    except Exception as e:
        results[prompt] = f"Error: {e}"

# Save results
with open("results.json", "w") as f:
    json.dump(results, f, indent=2)


Example B: Model Performance Tracking
--------------------------------------
from core.model_fallback import ModelFallbackManager
from core.openrouter_checker import ApiKeyChecker
import time

api_key = "your_api_key_here"
checker = ApiKeyChecker(api_key)
fallback = checker.fallback_manager

# Track model performance
models = [
    "nvidia/nemotron-3-ultra-550b-a55b:free",
    "nvidia/nemotron-3-super-120b-a12b:free"
]

for model in models:
    start = time.time()
    try:
        response = checker.chat_completion(model, "Hello")
        elapsed = time.time() - start
        print(f"{model}: {elapsed:.2f}s")
    except Exception as e:
        print(f"{model}: Failed - {e}")

# Show status
status = fallback.get_status()
print(f"Fallback status: {status}")
